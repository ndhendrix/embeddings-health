Question 3: How much variability is there in embeddings across census tracts for a given year?

For this, I’ll need to combine block group embeddings via an area-weighted average into census tracts. The ID for census tracts is equivalent to the first 11 digits of the FIPS code for block groups.

Code 
# Create census tract ID (first 11 digits of block group FIPS)
embeddings_df = embeddings_df.with_columns(
    pl.col("census_block_group").cast(pl.Utf8).str.slice(0, 11).alias("census_tract")
)

# Compute area-weighted average embeddings at the tract level for a given year
year_to_plot = 2024
year_data = embeddings_df.filter(pl.col("embedding_year") == year_to_plot)

# Area-weighted average: for each tract, weight each block group's embedding by its area
tract_embeddings = (
    year_data
    .with_columns([
        (pl.col(c) * pl.col("area_km2")).alias(f"{c}_weighted") for c in emb_cols
    ])
    .group_by("census_tract")
    .agg(
        [pl.col(f"{c}_weighted").sum().alias(c) for c in emb_cols]
        + [pl.col("area_km2").sum().alias("total_area")]
        + [pl.col("state_fips").first()]
    )
    .with_columns([
        (pl.col(c) / pl.col("total_area")).alias(c) for c in emb_cols
    ])
    .with_columns(
        pl.col("state_fips").replace_strict(STATE_FIPS_TO_DIVISION, default="Unknown").alias("census_division")
    )
)

# PCA on tract-level embeddings
tract_emb_matrix = tract_embeddings.select(emb_cols).to_numpy()
pca_tract = PCA(n_components=2, random_state=42)
tract_coords = pca_tract.fit_transform(tract_emb_matrix)

tract_embeddings = tract_embeddings.with_columns(
    pl.Series("pc1", tract_coords[:, 0]),
    pl.Series("pc2", tract_coords[:, 1]),
)

fig, ax = plt.subplots(figsize=(12, 9))

divisions = sorted(tract_embeddings["census_division"].unique().to_list())
cmap = plt.cm.tab20(np.linspace(0, 1, len(divisions)))

for i, div in enumerate(divisions):
    div_data = tract_embeddings.filter(pl.col("census_division") == div)
    ax.scatter(
        div_data["pc1"].to_numpy(),
        div_data["pc2"].to_numpy(),
        c=[cmap[i]],
        label=div,
        alpha=0.6,
        s=20,
        edgecolors="none",
    )

ax.set_xlabel(f"PC1 ({pca_tract.explained_variance_ratio_[0]:.1%} var)")
ax.set_ylabel(f"PC2 ({pca_tract.explained_variance_ratio_[1]:.1%} var)")
ax.set_title(
    f"PCA of Census Tract Embeddings (Area-Weighted) — {year_to_plot}\nColored by Census Division",
    fontsize=13, fontweight="bold"
)
ax.legend(fontsize=8, markerscale=2, title="Census Division", bbox_to_anchor=(1.02, 1), loc="upper left")
plt.tight_layout()
plt.show()

print(f"Number of tracts: {tract_embeddings.shape[0]}")
print(f"Variance explained: PC1={pca_tract.explained_variance_ratio_[0]:.1%}, PC2={pca_tract.explained_variance_ratio_[1]:.1%}")

Please plot the distribution of each health outcome across tracts within 2025 using a violin plot or similar.

Note: The PLACES 2025 release contains data from years 2022–2023. Using the most recent year available (2023).

Code 
# Load PLACES data
places_df = pl.read_csv(
    DATA_DIR / "PLACES__Local_Data_for_Better_Health__Census_Tract_Data__2025_release.csv",
    infer_schema_length=5000,
)

# Filter to most recent year and crude prevalence
places_latest = places_df.filter(
    (pl.col("Year") == pl.col("Year").max()) &
    (pl.col("DataValueTypeID") == "CrdPrv")
)

# Get unique measures
measures = sorted(places_latest["Short_Question_Text"].unique().to_list())
print(f"Number of health outcome measures: {len(measures)}")
print(f"Data year: {places_latest['Year'].unique().to_list()}")

# Pivot to wide format: one row per tract, one column per measure
places_wide = places_latest.pivot(
    on="Short_Question_Text",
    index="LocationID",
    values="Data_Value",
).drop_nulls()

# Create violin plots - split into multiple figures for readability
n_measures = len(measures)
n_per_fig = 10
n_figs = (n_measures + n_per_fig - 1) // n_per_fig

for fig_idx in range(n_figs):
    start = fig_idx * n_per_fig
    end = min(start + n_per_fig, n_measures)
    fig_measures = measures[start:end]

    fig, ax = plt.subplots(figsize=(14, max(5, len(fig_measures) * 0.7)))

    data_arrays = []
    labels = []
    for m in fig_measures:
        if m in places_wide.columns:
            vals = places_wide[m].drop_nulls().to_numpy()
            if len(vals) > 0:
                data_arrays.append(vals)
                labels.append(m)

    if data_arrays:
        parts = ax.violinplot(data_arrays, vert=False, showmedians=True, showextrema=True)

        # Style the violins
        for pc in parts["bodies"]:
            pc.set_facecolor("#5B8DEF")
            pc.set_alpha(0.7)
        parts["cmedians"].set_color("#E74C3C")
        parts["cmedians"].set_linewidth(2)

        ax.set_yticks(range(1, len(labels) + 1))
        ax.set_yticklabels(labels, fontsize=9)
        ax.set_xlabel("Crude Prevalence (%)", fontsize=11)
        ax.set_title(
            f"Distribution of Health Outcomes Across Census Tracts (PLACES 2025 Release)\n"
            f"Part {fig_idx + 1}/{n_figs}",
            fontsize=12, fontweight="bold",
        )
        ax.grid(axis="x", alpha=0.3)

    plt.tight_layout()
    plt.show()

from sklearn.linear_model import Ridge
from sklearn.metrics import r2_score, mean_absolute_error

# --- Compute 2022 tract-level embeddings for temporal alignment ---
# ReADI is 2022, so we use 2022 embeddings and 2022 PLACES data
pred_year = 2022
year_data_2022 = embeddings_df.filter(pl.col("embedding_year") == pred_year)

tract_emb_2022 = (
    year_data_2022
    .with_columns([(pl.col(c) * pl.col("area_km2")).alias(f"{c}_weighted") for c in emb_cols])
    .group_by("census_tract")
    .agg(
        [pl.col(f"{c}_weighted").sum().alias(c) for c in emb_cols]
        + [pl.col("area_km2").sum().alias("total_area")]
        + [pl.col("state_fips").first()]
    )
    .with_columns([(pl.col(c) / pl.col("total_area")).alias(c) for c in emb_cols])
)

# Compute 10 PCs from 2022 tract-level embeddings
pca_10 = PCA(n_components=10, random_state=42)
tract_pcs = pca_10.fit_transform(tract_emb_2022.select(emb_cols).to_numpy())

tract_pc_df = tract_emb_2022.select("census_tract", "state_fips").with_columns(
    [pl.Series(f"pc{i}", tract_pcs[:, i]) for i in range(10)]
)
pc_cols = [f"pc{i}" for i in range(10)]

print(f"Using {pred_year} embeddings ({tract_emb_2022.shape[0]} tracts)")
print(f"Variance explained by 10 PCs: {pca_10.explained_variance_ratio_.sum():.1%}")

# --- State-level train/test split to avoid spatial leakage ---
all_states = sorted(tract_pc_df["state_fips"].unique().to_list())
rng_states = np.random.default_rng(42)
n_holdout = max(1, round(len(all_states) * 0.2))
holdout_states = set(rng_states.choice(all_states, size=n_holdout, replace=False))
train_states = set(all_states) - holdout_states

print(f"Total states: {len(all_states)}")
print(f"Holdout states ({len(holdout_states)}): {sorted(holdout_states)}")
print(f"Train states ({len(train_states)}): {sorted(train_states)}")

# Join with ReADI data on census_tract
merged_readi = tract_pc_df.join(
    readi_clean.select("census_tract", "readi_national_rank"),
    on="census_tract",
    how="inner",
)

train_mask = merged_readi["state_fips"].is_in(list(train_states))
test_mask = merged_readi["state_fips"].is_in(list(holdout_states))

X_train = merged_readi.filter(train_mask).select(pc_cols).to_numpy()
y_train = merged_readi.filter(train_mask)["readi_national_rank"].to_numpy()
X_test = merged_readi.filter(test_mask).select(pc_cols).to_numpy()
y_test = merged_readi.filter(test_mask)["readi_national_rank"].to_numpy()

print(f"\nReADI — Train tracts: {len(y_train)}, Test tracts (held-out states): {len(y_test)}")

# Train on in-sample states, predict on held-out states
model = Ridge(alpha=1.0)
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

r2 = r2_score(y_test, y_pred)
mae = mean_absolute_error(y_test, y_pred)

# Plot predicted vs observed
fig, ax = plt.subplots(figsize=(8, 8))
ax.scatter(y_test, y_pred, alpha=0.3, s=8, color="#5B8DEF", edgecolors="none")

lims = [min(y_test.min(), y_pred.min()), max(y_test.max(), y_pred.max())]
ax.plot(lims, lims, "--", color="#E74C3C", linewidth=1.5, label="Perfect prediction")

ax.set_xlabel("Observed ReADI National Rank", fontsize=12)
ax.set_ylabel("Predicted ReADI National Rank", fontsize=12)
ax.set_title(
    f"Predicting ReADI from Spatial Embeddings (10 PCs)\n"
    f"State-Level Holdout: R² = {r2:.3f}, MAE = {mae:.1f}\n"
    f"({len(holdout_states)} held-out states, {len(y_test)} test tracts)",
    fontsize=12, fontweight="bold",
)
ax.legend(fontsize=10)
ax.set_aspect("equal")
plt.tight_layout()
plt.show()

X_test_h = test_health.select(pc_cols).to_numpy()

# ReADI features (single column)
readi_train = train_health["readi_national_rank"].to_numpy().astype(float)
readi_test = test_health["readi_national_rank"].to_numpy().astype(float)

print(f"Train tracts: {train_health.shape[0]}, Test tracts: {test_health.shape[0]}")

results = []
for measure in health_measures:
    y_train_m = train_health[measure].to_numpy().astype(float)
    y_test_m = test_health[measure].to_numpy().astype(float)

    # Need valid rows for both outcome AND ReADI
    train_valid = ~np.isnan(y_train_m) & ~np.isnan(readi_train)
    test_valid = ~np.isnan(y_test_m) & ~np.isnan(readi_test)
    if train_valid.sum() < 50 or test_valid.sum() < 10:
        continue
    if np.std(y_train_m[train_valid]) < 1e-6:
        continue

    # --- Embedding-based prediction ---
    model_emb = Ridge(alpha=1.0)
    model_emb.fit(X_train_h[train_valid], y_train_m[train_valid])
    y_pred_emb = model_emb.predict(X_test_h[test_valid])
    y_obs_m = y_test_m[test_valid]

    r2_emb = r2_score(y_obs_m, y_pred_emb)
    mae_emb = mean_absolute_error(y_obs_m, y_pred_emb)

    # --- ReADI-based prediction (baseline) ---
    model_readi = Ridge(alpha=1.0)
    model_readi.fit(readi_train[train_valid].reshape(-1, 1), y_train_m[train_valid])
    y_pred_readi = model_readi.predict(readi_test[test_valid].reshape(-1, 1))
    r2_readi = r2_score(y_obs_m, y_pred_readi)

    results.append({
        "measure": measure,
        "r2": r2_emb,
        "mae": mae_emb,
        "r2_readi": r2_readi,
        "y_obs": y_obs_m,
        "y_pred": y_pred_emb,
    })

# Sort by R²
results.sort(key=lambda x: x["r2"], reverse=True)

# Print all R² values with ReADI comparison
print(f"\n--- All Health Outcomes ranked by R² (state-level holdout) ---")
print(f"  {'Measure':40s}  {'Emb R²':>8s}  {'ReADI R²':>8s}")
for r in results:
    print(f"  {r['measure']:40s}  {r['r2']:8.3f}  {r['r2_readi']:8.3f}")
